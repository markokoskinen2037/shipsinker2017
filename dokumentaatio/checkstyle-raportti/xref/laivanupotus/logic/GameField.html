14  /**
15   * Luokka tarjoaa laivojen asetteluun ja pelin kirjanpitoon käytettäviä
16   * metodeja.
17   */
18  public class GameField {
19  
20      private final Game gui;
21      private int shipsLeft;
22      private ArrayList<Ship> shipList;
23      private boolean running;
24  
25      /**
26       * Konstruktori tallentaa saamansa parametrit omiin private muuttujiinsa,
27       * luo tyhjän listan laivoille ja aloittaa pelin tarkkailun.
28       *
29       * @param game Main metodin luoma game olio
30       * @param numberOfShips Main metodin kertoma laivojen määrä kartalla
31       * aloitustilanteessa
32       */
33      public GameField(Game game, int numberOfShips) { //10 laivaa kartalla
34  
35          this.shipsLeft = numberOfShips;
36          this.gui = game;
37          this.shipList = new ArrayList<>();
38          checkForShipsEverySecond();
39  
40      }
41  
42      public Game getGui() {
43          return gui;
44      }
45  
46      public int getShipsLeft() {
47          return shipsLeft;
48      }
49  
50      public void setShipsLeft(int amount) {
51          this.shipsLeft = amount;
52      }
53  
54      public ArrayList<Ship> getShipList() {
55          return shipList;
56      }
57  
58      /**
59       * Lisää parametrina saadun Ship alkion koordinaatteja vastaaviin
60       * JButtoneihin ActionListener tapahtumankuuntelijat joiden avulla GameField
61       * huomaa kun joku laiva tai sen osa tuhoutuu.
62       *
63       * @param ship Main metodin antama parametri
64       */
65      public void addShipToGui(Ship ship) {
66  
67          shipList.add(ship);
68  
69          for (Object ship1Cordinate : ship.getCordinates()) {
70              //System.out.println("Found correct JButton element called: " + gui.getComponentByName(ship1Cordinate.toString()).getName());
71              JButton targetButton = gui.getComponentByName(ship1Cordinate.toString());
72  
73              ActionListener shipListener = new ActionListener() {
74                  @Override
75                  public void actionPerformed(ActionEvent ae) {
76                      targetButton.setBackground(Color.black);
77                      targetButton.setEnabled(false);
78                      ship.destroyPart();
79                      if (ship.isDestroyed()) {
80                          shipsLeft -= 1;
81                      }
82                  }
83              };
84              targetButton.addActionListener(shipListener);
85          }
86  
87      }
88  
89      private void checkForShipsEverySecond() { //Checks how many ships are still left every second
90          this.running = true;
91  
92          Runnable checkRemainingShips = new Runnable() {
93              int gametime = 0;
94  
95              public void run() {
96  
97                  if (getShipsLeft() >= 1) {
98                      System.out.println("ships: " + getShipsLeft() + " | time:" + gametime);
99                      gametime++;
100                 }
101 
102                 if (getShipsLeft() == 0 && running) {
103                     gui.setVisible(false);
104                     System.out.println("You beat the game in " + gametime + " seconds!");
105                     JOptionPane.showMessageDialog(gui, "You beat the game in " + gametime + " seconds!");
106                     running = false;
107                 }
108 
109             }
110         };
111 
112         ScheduledExecutorService executor = Executors.newScheduledThreadPool(1);
113         executor.scheduleAtFixedRate(checkRemainingShips, 0, 1, TimeUnit.SECONDS);
114     }
115 
116     /**
117      * Palauttaa true jos peli on käynnissä ja false jos peli ei ole käynnissä.
118      *
119      * @return boolean
120      */
121     public boolean isRunning() {
122         return this.running;
123     }
124 }
